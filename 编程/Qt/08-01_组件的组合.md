# 组件的组合

以下是修改后的代码，演示如何在 `main` 函数中手动创建组件，而不是使用 Qt Designer 生成的 UI 类：

```cpp
#include "mainwindow.h"
#include <QApplication>
#include <QVBoxLayout>
#include <QLabel>
#include <QPushButton> // 添加QPushButton头文件

int main(int argc, char *argv[])
{
    QApplication a(argc, argv);
    QMainWindow Window;//创建菜单窗口
    Window.setWindowTitle("带按钮的窗口"); // 设置窗口标题
    Window.resize(400, 300); // 设置窗口大小
    
    QWidget *centralWidget = new QWidget(&Window);//通用容器
    Window.setCentralWidget(centralWidget);//指定通用容器为布局结构中的中心区域

    // 创建布局
    QVBoxLayout *layout = new QVBoxLayout(centralWidget);//垂直布局
    layout->setSpacing(10); // 设置控件间距
    layout->setContentsMargins(20, 20, 20, 20); // 设置边距

    // 创建标签组件
    QLabel *label1 = new QLabel("输入内容1", centralWidget);
    QLabel *label2 = new QLabel("输入内容2", centralWidget);
    
    // 创建按钮组件
    QPushButton *button1 = new QPushButton("确定", centralWidget); // 创建确定按钮
    QPushButton *button2 = new QPushButton("取消", centralWidget); // 创建取消按钮
    
    // 设置按钮属性（可选）
    button1->setDefault(true); // 设置默认按钮（按回车触发）
    button2->setToolTip("点击此按钮取消操作"); // 设置工具提示

    // 添加布局
    layout->addWidget(label1);
    layout->addWidget(label2);
    layout->addStretch(1); // 添加弹性空间，将按钮推到底部
    layout->addWidget(button1);
    layout->addWidget(button2);

    // 连接按钮信号到槽函数（可选）
    QObject::connect(button1, &QPushButton::clicked, []() {
        qDebug() << "确定按钮被点击";
        // 这里可以添加按钮点击后的逻辑
    });
    
    QObject::connect(button2, &QPushButton::clicked, []() {
        qDebug() << "取消按钮被点击";
        // 这里可以添加按钮点击后的逻辑
    });

    Window.show();
    return a.exec();
}
```
