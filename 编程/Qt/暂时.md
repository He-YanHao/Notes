```cpp
    // 创建控件并统一设置属性
    auto createInputField = [](const QString &text) {
        QLineEdit *field = new QLineEdit(text);
        field->setFixedHeight(50);
        field->setStyleSheet("font-size: 16px;");  // 统一字体大小
        return field;
    };
```
